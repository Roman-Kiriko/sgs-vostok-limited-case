CREATE TABLE "CitiesTable" (
	"id"	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
	"Name"	TEXT NOT NULL
)

CREATE TABLE "GuildsTable" (
	"id"	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
	"Name"	TEXT NOT NULL,
	"CityId"	INTEGER NOT NULL,
	FOREIGN KEY("CityId") REFERENCES "CitiesTable"("id") ON UPDATE NO ACTION ON DELETE CASCADE
)

CREATE TABLE "EmployeesTable" (
	"id"	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
	"Name"	TEXT NOT NULL,
	"GuildId"	INTEGER NOT NULL,
	FOREIGN KEY (GuildId) REFERENCES GuildsTable (id) ON UPDATE NO ACTION ON DELETE CASCADE
)

CREATE TABLE "BrigadesTable" (
	"id"	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
	"EmployeeId"	INTEGER NOT NULL,
	"ShiftId"	INTEGER NOT NULL,
	FOREIGN KEY (EmployeeId) REFERENCES EmployeesTable (id) ON UPDATE NO ACTION ON DELETE CASCADE,
	FOREIGN KEY (ShiftId) REFERENCES ShiftsTable (id) ON UPDATE NO ACTION ON DELETE CASCADE
)

CREATE TABLE "ShiftsTable" (
	"id"	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
	"Name"	TEXT NOT NULL
)
